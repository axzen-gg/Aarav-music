-- ── PROTECTION ENGINE ──
local ProtectionStates = {AntiSit=false, Noclip=false, SuperNoclip=false, AntiFling=false, Freeze=false}
local Cons = {}
local function toggleProtection(feature, state)
    ProtectionStates[feature]=state
    local char=LocalPlayer.Character
    local hum=char and char:FindFirstChildOfClass("Humanoid")
    local root=char and char:FindFirstChild("HumanoidRootPart")
    if Cons[feature] then Cons[feature]:Disconnect(); Cons[feature]=nil end
    if feature=="AntiSit" and state then
        if hum and hum.Sit then hum.Sit=false end
        Cons.AntiSit=RunService.Stepped:Connect(function()
            local c=LocalPlayer.Character; local h=c and c:FindFirstChildOfClass("Humanoid")
            if h and h.Sit then h.Sit=false end
        end)
    elseif feature=="Noclip" and state then
        Cons.Noclip=RunService.Stepped:Connect(function()
            if LocalPlayer.Character then
                for _,v in pairs(LocalPlayer.Character:GetDescendants()) do
                    if v:IsA("BasePart") and v.CanCollide then
                        if not string.find(v.Name:lower(),"leg") and not string.find(v.Name:lower(),"foot") then
                            v.CanCollide=false
                        end
                    end
                end
            end
        end)
    elseif feature=="SuperNoclip" and state then
        Cons.SuperNoclip=RunService.Stepped:Connect(function()
            if LocalPlayer.Character then
                for _,v in pairs(LocalPlayer.Character:GetDescendants()) do
                    if v:IsA("BasePart") and v.CanCollide then v.CanCollide=false end
                end
            end
        end)
    elseif feature=="AntiFling" and state then
        Cons.AntiFling=RunService.Stepped:Connect(function()
            for _,p in pairs(Players:GetPlayers()) do
                if p~=LocalPlayer and p.Character then
                    for _,v in pairs(p.Character:GetDescendants()) do
                        if v:IsA("BasePart") then v.CanCollide=false end
                    end
                end
            end
            local myRoot=LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if myRoot and (myRoot.Velocity.Magnitude>250 or myRoot.RotVelocity.Magnitude>250) then
                myRoot.Velocity=Vector3.new(0,0,0); myRoot.RotVelocity=Vector3.new(0,0,0)
            end
        end)
    elseif feature=="Freeze" then
        if root then root.Anchored=state end
        if state then
            Cons.Freeze=LocalPlayer.CharacterAdded:Connect(function(newChar)
                local newRoot=newChar:WaitForChild("HumanoidRootPart",5)
                if newRoot then newRoot.Anchored=true end
            end)
        end
    end
end

        if p and p.Character then return p.Character:FindFirstChild("HumanoidRootPart") end
    end
    return nil
end
local function stopPropEngine() activeMode="None"; engineLoopActive=false end
local function startPropEngine(mode, targetName, visType)
    stopPropEngine(); task.wait(0.1)
    activeMode=mode; engineLoopActive=true
    task.spawn(function()
        while engineLoopActive do
            local root=getTargetRoot(targetName)
            local props=getMyProps()
            if root and #props>0 then
                local t=tick()
                for i,prop in ipairs(props) do
                    task.spawn(function()
                        local base=root.CFrame
                        local final=base
                        local col=Color3.fromHSV((t*0.5+(i/#props))%1,1,1)
                        if activeMode=="Nuke" then
                            final=base*CFrame.new(math.random(-2,2),math.random(0,3),math.random(-2,2))*CFrame.Angles(math.random(),math.random(),math.random())
                        elseif activeMode=="Visualizer" then
                            if visType=="Orbital" then
                                local angle=(t*3)+((i/#props)*math.pi*2)
                                final=base*CFrame.new(math.cos(angle)*6,0,math.sin(angle)*6)
                            elseif visType=="Tornado" then
                                local angle=(t*5)+((i/#props)*math.pi*2)
                                local h=(i/#props)*12; local r=2+(i/#props)*5
                                final=base*CFrame.new(math.cos(angle)*r,h-4,math.sin(angle)*r)
                            elseif visType=="Swarm" then
                                local ox=math.noise(t,i,0)*8; local oy=math.noise(t,0,i)*6; local oz=math.noise(0,t,i)*8
                                final=base*CFrame.new(ox,oy+2,oz)
                            elseif visType=="Grid" then
                                local sp=5; local cols=math.ceil(math.sqrt(#props))
                                local row=math.floor((i-1)/cols); local col2=(i-1)%cols
                                final=base*CFrame.new((col2-cols/2)*sp,0,(row-cols/2)*sp)
                            end
                        end
                        pcall(function()
                            prop.SetCurrentCFrame:InvokeServer(final)
                            if prop:FindFirstChild("ChangePropColor") then prop.ChangePropColor:InvokeServer(col) end
                        end)
                    end)
                end
            end
            task.wait(0.08)
        end
    end)
end

local currentShuffleSong = nil
local function buildShuffleUI(p)
    mkLabel(p,"🎲 Random Shuffle Mix",4,C.GOLD,13)
    local shuffleBtn = mkBtn(p,"🔀  Play Random Mix",C.PURPLE,28,32)
    local nowLbl = Instance.new("TextLabel")
    nowLbl.Size=UDim2.new(1,-4,0,50); nowLbl.Position=UDim2.new(0,2,0,68)
    nowLbl.BackgroundTransparency=1; nowLbl.Text="Click Shuffle to start!"
    nowLbl.TextColor3=C.GRAY; nowLbl.Font=Enum.Font.Gotham; nowLbl.TextSize=10
    nowLbl.TextWrapped=true; nowLbl.ZIndex=4; nowLbl.Parent=p

    local function PlayRandom()
        if #AllSongs==0 then return end
        local s=AllSongs[math.random(1,#AllSongs)]
        currentShuffleSong=s
        nowLbl.Text="🎵 Now Playing:\n"..s[1].."\nID: "..s[2]
        local char=LocalPlayer.Character; local onVeh=false
        if char then for _,v in pairs(char:GetChildren()) do
            if v.Name=="SegwaySmall" or v.Name=="SkateBoard" then onVeh=true; break end
        end end
        if onVeh then if VehicleRemote then StartRGBLoop(); ForcePlayMusic(VehicleRemote,tostring(s[2])) end
        else SmartPlayMusic(tostring(s[2]),SelectedVehicle,false) end
    end
    shuffleBtn.MouseButton1Click:Connect(PlayRandom)

    local nextBtn = mkBtn(p,"⏭  Next Song",C.BLUE,128,28)
    nextBtn.MouseButton1Click:Connect(PlayRandom)

    local saveBtn = mkBtn(p,"💾  Save Current",C.GOLD,164,28)
    saveBtn.MouseButton1Click:Connect(function()
        if currentShuffleSong and #SavedPlaylist<MaxSavedSongs then
            table.insert(SavedPlaylist,{name=currentShuffleSong[1],id=tostring(currentShuffleSong[2])})
            SavePlaylistToFile(); buildSavedUI(SavedPage)
        end
    end)
end

-- Collision handling
task.spawn(function()
    local function setupCollision(character)
        local hrp=character:WaitForChild("HumanoidRootPart")
        local bodyParts={}; hrp.CanCollide=true
        local function addPart(part)
            if part:IsA("BasePart") and part~=hrp then table.insert(bodyParts,part) end
        end
        for _,part in pairs(character:GetDescendants()) do addPart(part) end
        local dc=character.DescendantAdded:Connect(addPart)
        local hb; hb=RunService.Heartbeat:Connect(function()
            if not character or not character.Parent then hb:Disconnect(); dc:Disconnect(); return end
            for _,pt in ipairs(bodyParts) do if pt and pt.Parent then pt.CanCollide=false end end
        end)
    end
    if LocalPlayer.Character then setupCollision(LocalPlayer.Character) end
    LocalPlayer.CharacterAdded:Connect(setupCollision)
end)
